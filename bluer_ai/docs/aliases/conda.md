# conda

```bash
@conda \
	create \
	[clone=<auto|base>,~install_plugin,name=<environment-name>,repo=<repo-name>,~recreate]
 . create conda environment.
@conda \
	exists \
	[name=<environment-name>]
 . does conda environment exist?
@conda \
	list \
	[dryrun]
 . show list of conda environments.
@conda \
	rm \
	[dryrun,name=<environment-name>]
 . rm conda environment.
```
